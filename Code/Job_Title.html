# Word Cloud for Job Titles
# import libraries
import matplotlib.pyplot as plt
from wordcloud import WordCloud, STOPWORDS

# create a list of job titles
job_titles = df['title'].tolist()

# remove some words from job titles
job_titles = [title.replace('Machine Learning', '') for title in job_titles]
job_titles = [title.replace('Data Science', '') for title in job_titles]

# create a string of job titles
job_titles_string = ' '.join(job_titles)

# Mask for word cloud
import numpy as np
from PIL import Image

# create a mask
mask = np.array(Image.open('../Mini-Data-Viz-Project/images/wordcloud_mask.webp'))

# create a word cloud as ellipse
wordcloud = WordCloud(stopwords=STOPWORDS, background_color='white', width=2400, height=2000, mask=mask).generate(job_titles_string)

# plot the word cloud
plt.figure(figsize=(160,200))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis('off')
plt.show()
